161 bool Tache::verifierPrecedence(Tache &t) const
162 {
163  // On vérifie que l'on ajoute pas une même tâche en précédence d'elle-même
164  if(this == &t){
165  qDebug()<<"Je retourne faux";
166  return false;
167  }
168  // On vérifie que les dates d'échéances correspondent
169  if(t.getEcheance() > this->getEcheance()){
170  qDebug()<<"Je retourne faux";
171  return false;
172  }
173  // On vérifie que t ne fait pas déjà partie des précédences existantes
174  for(std::vector<Tache*>::const_iterator it_prece = this->getPrecedence().begin(); it_prece != this->getPrecedence().end(); ++it_prece)
175  if((*it_prece)->getIdentificateur() == t.getIdentificateur()){
176  qDebug()<<"Je retourne faux";
177  return false;
178  }
179 
180  // Vérifier pour les taches composites
181  TacheComposite* Compo = dynamic_cast<TacheComposite*>(&t);
182 
183  if(Compo){
184  for(std::vector<Tache*>::const_iterator it = Compo->getComposition().begin(); it != Compo->getComposition().end(); ++it){
185  if(!verifierPrecedence(**it))
186  return false;
187  }
188  }
189 
190  // On vérifie la même chose pour les taĉhes précédentes mères de this et en prenant compte du cas des compositions
191  std::vector<Tache*> tache_mere = this->getMere();
192 
193  for(std::vector<Tache*>::const_iterator it = tache_mere.begin(); it != tache_mere.end(); ++it){
194  TacheComposite* item = dynamic_cast<TacheComposite*>(*it);
195 
196  if(item){
197  for(std::vector<Tache*>::const_iterator it_compo = item->getComposition().begin(); it_compo != item->getComposition().end(); ++it_compo)
198  if(!(*it)->verifierPrecedence(t))

203  return false;
