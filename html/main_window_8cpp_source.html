194 void MainWindow::MAJ_treeview_Composition()
195 {
196  ui->TreeWidget->clear();
197  ProjetManager& projetmanager = ProjetManager::getInstance();
198  vector<QTreeWidgetItem*> afficher;
199 
200  for(vector<Projet *>::const_iterator it_projets = projetmanager.getProjets().begin(); it_projets != projetmanager.getProjets().end(); ++it_projets){
201  QTreeWidgetItem *treeItem = addTreeRoot((*it_projets)->getId(), (*it_projets)->getTitre());
202  afficher.push_back(treeItem);
203  vector<Tache*> A_afficher = (*it_projets)->getTaches();
204  vector<Tache*>::iterator it_taches = A_afficher.begin();
205  // Affichage des tâches directement reliées au projet
206  while(it_taches != A_afficher.end()){
207  if((*it_taches)->getPere() == 0){
208  afficher.push_back(addTreeChild(treeItem, (*it_taches)->getIdentificateur(), (*it_taches)->getTitre()));
209  it_taches = A_afficher.erase(it_taches);
210  } else {
211  ++it_taches;
212  }
213  }
214 
215  // Affichage des compositions
216  it_taches = A_afficher.begin();
217  while(!A_afficher.empty()){
218  QTreeWidgetItem* item = trouver(afficher, (*it_taches)->getPere()->getIdentificateur());
219  if(item){
220  afficher.push_back(addTreeChild(item, (*it_taches)->getIdentificateur(), (*it_taches)->getTitre()));
221  it_taches = A_afficher.erase(it_taches);
222  } else {
223  ++it_taches;
224  }
225  if(it_taches == A_afficher.end())
226  it_taches = A_afficher.begin();
227  }
228  }
